       private detectBackend(): Backend {

             console.debug("[xclip] Selected wl-clipboard backend for Wayland");
             return "wl-clipboard";
           }
           if (isToolAvailable("xclip")) {
             console.debug("[xclip] Warning: Wayland detected but wl-copy not found, falling back to xclip (XWayland)");
             console.debug("[xclip] For best Wayland support, install wl-clipboard: apt install wl-clipboard");
             return "xclip";
           }
           throw new Error("No clipboard tool available. Install wl-clipboard (recommended for Wayland) or xclip.");
         }
         
         // X11
         if (isToolAvailable("xclip")) {
           console.debug("[xclip] Selected xclip backend for X11");
           return "xclip";
         }
         throw new Error("xclip not installed. Install with: apt install xclip (or pacman -S xclip)");
       }
